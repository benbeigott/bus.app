        function getMonday(date) {
            const d = new Date(date);
            const day = d.getDay();
            const diff = d.getDate() - day + (day === 0 ? -6 : 1);
            return new Date(d.setDate(diff));
        }

        function formatDate(date) {
            return date.toISOString().split('T')[0];
        }

        function formatDateDisplay(date) {
            return date.toLocaleDateString('de-DE', { day: '2-digit', month: '2-digit' });
        }

        function formatDateLong(date) {
            return date.toLocaleDateString('de-DE', { weekday: 'long', day: 'numeric', month: 'long' });
        }

        function getWeekNumber(date) {
            const d = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate()));
            const dayNum = d.getUTCDay() || 7;
            d.setUTCDate(d.getUTCDate() + 4 - dayNum);
            const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
            return Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
        }

        function persist() {
            localStorage.setItem('busstore_fleet', JSON.stringify(state.fleet));
            localStorage.setItem('busstore_bookings', JSON.stringify(state.bookings));

            const toast = document.getElementById('toast');
            const toastMessage = document.getElementById('toastMessage');
            const toastIcon = document.getElementById('toastIcon');
            
            toastMessage.textContent = message;
            toastIcon.textContent = type === 'error' ? '✕' : '✓';
            toastIcon.className = `text-lg ${type === 'error' ? 'text-red-400' : 'text-gold'}`;
            
            toast.classList.add('show');
            setTimeout(() => toast.classList.remove('show'), 3000);
        }

        function generateCode(length = 8) {
            const chars = 'abcdefghijklmnopqrstuvwxyz0123456789';
            let result = '';
            for (let i = 0; i < length; i++) {
                result += chars.charAt(Math.floor(Math.random() * chars.length));
            }
            return result;
        }

                showToast('Account wurde deaktiviert', 'error');

            sessionStorage.setItem('busstore_user', JSON.stringify(partner));

            showToast(`Willkommen, ${partner.company}!`);
        }

            if (!state.currentPartnerId) return;

            state.partners = state.partners.filter(p => p.id !== state.currentPartnerId);
            persist();

            showToast('Partner gelöscht');

            selector.innerHTML = `
                <option value="">Alle Busse anzeigen</option>

            `;

        function selectBus(busId) {
            state.selectedBusId = busId || null;
            renderFleet();
            renderCalendar();
        }

        function renderFleet() {
            const list = document.getElementById('fleetList');
            
            list.innerHTML = state.fleet.map(bus => {
                const isSelected = bus.id === state.selectedBusId;
                const todayBookings = state.bookings.filter(b => 
                    b.busId === bus.id && 
                    b.date === formatDate(new Date()) && 
                    b.status !== 'cancelled'
                ).length;
                
                return `
                    <div 
                        onclick="selectBus('${isSelected ? '' : bus.id}')"
                        class="p-4 rounded-xl cursor-pointer transition ${
                            isSelected 
                                ? 'bg-gold/10 border border-gold/30' 
                                : 'bg-black-elevated border border-white/5 hover:border-white/10'
                        }"
                    >
                        <div class="flex justify-between items-start">
                            <div>
                                <p class="font-medium ${isSelected ? 'text-gold' : 'text-white'}">${bus.name}</p>
                                <p class="text-xs text-zinc-600 mt-1">${bus.plate || bus.id}</p>
                            </div>
                            <span class="text-xs text-zinc-500 bg-black-surface px-2 py-1 rounded">${bus.seats}P</span>
                        </div>
                        ${todayBookings > 0 ? `
                            <div class="mt-3 pt-3 border-t border-white/5 flex items-center gap-2">

                            </div>
                        ` : ''}
                    </div>
                `;
            }).join('');
        }

            document.getElementById('addBusModal').classList.add('hidden');

        function handleAddBus(e) {
            e.preventDefault();
            
            const name = document.getElementById('busName').value.trim();
            const seats = parseInt(document.getElementById('busSeats').value);
            const plate = document.getElementById('busPlate').value.trim();
            
            const bus = {
                id: `BUS-${Date.now()}`,
                name,
                seats,
                plate: plate || `BUS-${state.fleet.length + 1}`
            };
            
            state.fleet.push(bus);
            persist();
            
            closeAddBusModal();
            renderBusSelector();
            renderFleet();
            renderFleetGrid();
            renderCalendar();
            
            showToast(`${name} hinzugefügt`);
        }

        function deleteBus(busId) {
            const bus = state.fleet.find(b => b.id === busId);
            if (!bus) return;
            
            const hasBookings = state.bookings.some(b => b.busId === busId && b.status !== 'cancelled');
            
            if (hasBookings) {
                if (!confirm(`"${bus.name}" hat aktive Buchungen. Trotzdem löschen?`)) {
                    return;
                }
            }
            
            state.fleet = state.fleet.filter(b => b.id !== busId);
            persist();
            
            renderBusSelector();
            renderFleet();
            renderFleetGrid();
            renderCalendar();
            
            showToast(`${bus.name} entfernt`);
        }

        function updateStats() {
            const weekStart = state.currentWeekStart;
            const weekEnd = new Date(weekStart);
            weekEnd.setDate(weekEnd.getDate() + 7);
            
            let weekBookings;
            if (state.currentUser.accessLevel === 'master') {
                weekBookings = state.bookings.filter(b => {
                    const d = new Date(b.date);
                    return d >= weekStart && d < weekEnd && b.status !== 'cancelled';
                });
            } else {
                weekBookings = state.bookings.filter(b => {
                    const d = new Date(b.date);
                    return d >= weekStart && d < weekEnd && b.status !== 'cancelled' && b.partnerId === state.currentUser.id;
                });
            }
            
            const totalRevenue = weekBookings.reduce((sum, b) => sum + b.price, 0);
            
            document.getElementById('statWeekBookings').textContent = weekBookings.length;
            document.getElementById('statWeekRevenue').textContent = `€${totalRevenue.toLocaleString()}`;
        }

        // ══════════════════════════════════════════════════════════
        // CALENDAR
        // ══════════════════════════════════════════════════════════

        function renderCalendar() {
            const grid = document.getElementById('calendarGrid');
            const days = [];
            const today = formatDate(new Date());
            
            for (let i = 0; i < 7; i++) {
                const date = new Date(state.currentWeekStart);
                date.setDate(date.getDate() + i);
                days.push(date);
            }
            
            const weekNum = getWeekNumber(state.currentWeekStart);
            const monthName = state.currentWeekStart.toLocaleDateString('de-DE', { month: 'long', year: 'numeric' });
            document.getElementById('weekDisplay').textContent = `Kalenderwoche ${weekNum}`;
            document.getElementById('dateRange').textContent = `${formatDateDisplay(days[0])} - ${formatDateDisplay(days[6])} · ${monthName}`;
            
            const busesToShow = state.selectedBusId 
                ? state.fleet.filter(b => b.id === state.selectedBusId)
                : state.fleet;
            
            let html = `
                <table class="w-full border-collapse">
                    <thead>
                        <tr>
                            <th class="w-48 p-3 text-left text-xs text-zinc-600 uppercase tracking-wider bg-black-surface border border-white/5">Bus</th>
                            ${days.map(date => {
                                const isToday = formatDate(date) === today;
                                return `
                                    <th class="p-3 text-center bg-black-surface border border-white/5 ${isToday ? 'bg-gold/5' : ''}">
                                        <p class="text-xs text-zinc-600 uppercase">${date.toLocaleDateString('de-DE', { weekday: 'short' })}</p>
                                        <p class="text-xl font-bold mt-1 ${isToday ? 'text-gold' : 'text-white'}">${date.getDate()}</p>
                                    </th>
                                `;
                            }).join('')}
                        </tr>
                    </thead>
                    <tbody>
            `;
            
            busesToShow.forEach(bus => {
                html += `
                    <tr>
                        <td class="p-3 bg-black-surface border border-white/5">
                            <p class="font-medium text-white text-sm">${bus.name}</p>
                            <p class="text-xs text-zinc-600">${bus.seats} Sitze · ${bus.plate}</p>
                        </td>
                        ${days.map(date => {
                            const dateStr = formatDate(date);
                            const isToday = dateStr === today;
                            const isPast = new Date(dateStr) < new Date(today);
                            const booking = state.bookings.find(b => 
                                b.busId === bus.id && 
                                b.date === dateStr && 
                                b.status !== 'cancelled'
                            );
                            
                            // Partner sieht nur eigene Buchungen als klickbar
                            const canViewBooking = state.currentUser.accessLevel === 'master' || 
                                (booking && booking.partnerId === state.currentUser.id);
                            
                            if (booking) {
                                return `
                                    <td 
                                        onclick="${canViewBooking ? `openBookingDetail('${booking.id}')` : ''}"
                                        class="calendar-cell booked ${isToday ? 'today' : ''} ${canViewBooking ? '' : 'cursor-default'}"
                                    >
                                        <p class="text-sm font-medium text-gold truncate">${booking.route.from}</p>
                                        <p class="text-xs text-zinc-500 truncate">→ ${booking.route.to}</p>
                                        <p class="text-sm font-bold text-white mt-2">€${booking.price.toLocaleString()}</p>
                                        ${state.currentUser.accessLevel === 'master' ? `
                                            <p class="text-xs text-zinc-600 mt-1">${booking.partnerName}</p>
                                        ` : ''}
                                    </td>
                                `;
                            }
                            
                            return `
                                <td 
                                    onclick="${isPast ? '' : `openBookingModal('${bus.id}', '${dateStr}')`}"
                                    class="calendar-cell ${isPast ? 'past' : ''} ${isToday ? 'today' : ''}"
                                >
                                    ${!isPast ? '<span class="text-xs text-zinc-700">+ Belegen</span>' : ''}
                                </td>
                            `;
                        }).join('')}
                    </tr>
                `;
            });
            
            html += '</tbody></table>';
            grid.innerHTML = html;
            
            updateStats();
        }

        function changeWeek(delta) {
            state.currentWeekStart.setDate(state.currentWeekStart.getDate() + (delta * 7));
            renderCalendar();
        }

        function goToToday() {
            state.currentWeekStart = getMonday(new Date());
            renderCalendar();
        }

        function openBookingModal(busId, date) {
            const bus = state.fleet.find(b => b.id === busId);
            const dateObj = new Date(date);
            
            document.getElementById('bookingBusId').value = busId;
            document.getElementById('bookingDate').value = date;
            document.getElementById('bookingModalSubtitle').textContent = 
                `${bus.name} · ${formatDateLong(dateObj)}`;
            
            document.getElementById('bookingForm').reset();
            document.getElementById('bookingBusId').value = busId;
            document.getElementById('bookingDate').value = date;
            
            // Master kann Partner auswählen
            const partnerSelect = document.getElementById('bookingPartnerSelect');
            const partnerDropdown = document.getElementById('bookingPartner');
            
            if (state.currentUser.accessLevel === 'master') {
                const activePartners = state.partners.filter(p => p.status === 'active');
                partnerDropdown.innerHTML = `
                    <option value="MASTER">👑 KING TOMEKKK (selbst)</option>
                    ${activePartners.map(p => `
                        <option value="${p.id}">${p.company}</option>
                    `).join('')}
                `;
                partnerSelect.classList.remove('hidden');
            } else {
                partnerSelect.classList.add('hidden');
            }
            
            document.getElementById('bookingModal').classList.remove('hidden');
        }

            document.getElementById('bookingModal').classList.add('hidden');
        }

        function handleBooking(e) {
            e.preventDefault();
            
            const busId = document.getElementById('bookingBusId').value;
            const date = document.getElementById('bookingDate').value;
            const from = document.getElementById('bookingFrom').value.trim();
            const to = document.getElementById('bookingTo').value.trim();
            const price = parseFloat(document.getElementById('bookingPrice').value);
            const notes = document.getElementById('bookingNotes').value.trim();
            
            const conflict = state.bookings.find(b => 
                b.busId === busId && 
                b.date === date && 
                b.status !== 'cancelled'
            );
            
            if (conflict) {
                showToast('Bus ist an diesem Tag bereits belegt', 'error');
                return;
            }
            
            let partnerId, partnerName;
            
            if (state.currentUser.accessLevel === 'master') {
                partnerId = document.getElementById('bookingPartner').value;
                if (partnerId === 'MASTER') {
                    partnerName = 'KING TOMEKKK';
                } else {
                    const partner = state.partners.find(p => p.id === partnerId);
                    partnerName = partner ? partner.company : 'Unbekannt';
                }
            } else {
                partnerId = state.currentUser.id;
                partnerName = state.currentUser.company;
            }
            
            const booking = {
                id: `BK-${Date.now()}`,
                busId,
                date,
                route: { from, to },
                price,
                notes: notes || null,
                partnerId,
                partnerName,
                createdAt: new Date().toISOString(),
                status: 'confirmed'
            };
            
            state.bookings.push(booking);
            persist();
            
            closeBookingModal();
            renderCalendar();
            renderFleet();
            
            showToast(`Fahrt gebucht: ${from} → ${to}`);
        }

        function openBookingDetail(bookingId) {
            const booking = state.bookings.find(b => b.id === bookingId);
            if (!booking) return;
            
            state.currentBookingId = bookingId;
            
            const bus = state.fleet.find(b => b.id === booking.busId);
            const dateObj = new Date(booking.date);
            
            document.getElementById('detailTitle').textContent = `${booking.route.from} → ${booking.route.to}`;
            document.getElementById('detailSubtitle').textContent = 
                `${bus?.name || 'Bus'} · ${formatDateLong(dateObj)}`;
            
            const statusEl = document.getElementById('detailStatus');
            statusEl.textContent = booking.status === 'confirmed' ? 'Bestätigt' : 'Storniert';
            statusEl.className = `status-badge ${booking.status === 'confirmed' ? 'status-active' : 'status-inactive'}`;
            
            document.getElementById('detailContent').innerHTML = `
                <div class="bg-black-elevated rounded-xl p-4 space-y-4">
                    <div class="flex justify-between items-center">
                        <span class="text-zinc-500 text-sm">Preis</span>
                        <span class="text-2xl font-bold text-gold">€${booking.price.toLocaleString()}</span>
                    </div>
                    <div class="h-px bg-white/5"></div>
                    <div class="flex justify-between">
                        <span class="text-zinc-500 text-sm">Partner</span>
                        <span class="text-white">${booking.partnerName}</span>
                    </div>
                    <div class="flex justify-between">
                        <span class="text-zinc-500 text-sm">Buchungs-ID</span>
                        <span class="text-zinc-400 font-mono text-sm">${booking.id}</span>
                    </div>
                    ${booking.notes ? `
                        <div class="h-px bg-white/5"></div>
                        <div>
                            <span class="text-zinc-500 text-sm block mb-2">Notizen</span>
                            <p class="text-white text-sm">${booking.notes}</p>
                        </div>
                    ` : ''}
                </div>
            `;
            
            const cancelBtn = document.getElementById('cancelBookingBtn');
            const canCancel = booking.status === 'confirmed' && 
                (state.currentUser.accessLevel === 'master' || booking.partnerId === state.currentUser.id);
            cancelBtn.style.display = canCancel ? 'block' : 'none';
            
            document.getElementById('bookingDetailModal').classList.remove('hidden');
        }

        function closeBookingDetailModal() {
            document.getElementById('bookingDetailModal').classList.add('hidden');
            state.currentBookingId = null;
        }

        function cancelCurrentBooking() {
            if (!state.currentBookingId) return;
            
            const booking = state.bookings.find(b => b.id === state.currentBookingId);
            if (booking) {
                booking.status = 'cancelled';
                persist();
                closeBookingDetailModal();
                renderCalendar();
                renderFleet();
                showToast('Buchung storniert');
            }
        }
